function debounce(func, wait) {
    let timeout;
    return function executedFunction(...args) {
        const later = () => {
            clearTimeout(timeout);
            func(...args);
        };
        clearTimeout(timeout);
        timeout = setTimeout(later, wait);
    };
}

function cargarSelectOptions(selectId, opciones) {

    while (select.options.length > 1) {

        opt.value = opcion;
        opt.textContent = opcion;

    });
}

function obtenerValoresFiltros() {
    const fechaDesde = document.getElementById('fechaDesde')?.value;
    const fechaHasta = document.getElementById('fechaHasta')?.value;
    
    return {
        fechaDesde: fechaDesde ? new Date(fechaDesde) : null,
        fechaHasta: fechaHasta ? new Date(fechaHasta


        + 'T23:59:59') : null,
        usuario: document.getElementById('filtroUsuario')?.value || '',
        tipoAccion: document.getElementById('filtroTipoAccion')?.value || '',
        lob: document.getElementById('filtroLob')?.value || '',
        busqueda: document.getElementById('busquedaRapida')?.value?.toLowerCase() || ''
    };
}

function cumpleFiltros(dato, filtros) {
    const fechaRegistro = new Date(dato.timestamp);
    
    const cumpleFecha = (!filtros.fechaDesde || fechaRegistro >= filtros.fechaDesde) && 
                       (!filtros.fechaHasta || fechaRegistro <= filtros.fechaHasta);
    
    const cumpleUsuario = !filtros.usuario || dato.usuario === filtros.usuario;
    const cumpleTipo = !filtros.tipoAccion || dato.tipoAccion === filtros.tipoAccion;
    const cumpleLob = !filtros.lob || dato.lob === filtros.lob;
    const cumpleBusqueda = !filtros.busqueda || 
                          (dato.bmsId && dato.bmsId.toString().toLowerCase().includes(filtros.busqueda)) ||
                          (dato.nombreAgente && dato.nombreAgente.toLowerCase().includes(filtros.busqueda));
    
    return cumpleFecha && cumpleUsuario && cumpleTipo && cumpleLob && cumpleBusqueda;
}

}

// TABLA AUDIT CON PAGINACI√ìN
// ========================================
function renderizarTabla() {
    const tbody = document.getElementById('auditTableBody');
    if (!tbody) return;
    
    const inicio = (paginaActual - 1) * registrosPorPagina;
    const fin = inicio + registrosPorPagina;
    const registrosPagina = datosFiltrados.slice(inicio, fin);
    
    if (registrosPagina.length === 0) {
        tbody.innerHTML = `
            <tr>
                <td colspan="9" style="text-align:center;padding:40px;color:#7f8c8d;">
                    No hay registros que coincidan con los filtros aplicados
                </td>
            </tr>
        `;
        actualizarInfoPaginacion(0);
        return;
    }
    
    tbody.innerHTML = registrosPagina.map(d => crearFilaTabla(d)).join('');
    actualizarInfoPaginacion(datosFiltrados.length);
}

function crearFilaTabla(dato) {
    const monto = dato.monto 
        ? `$${Number(dato.monto).toLocaleString('es-CO', {minimumFractionDigits: 2})}` 
        : '-';
    
    const datosEscapados = JSON.stringify(dato)
        .replace(/'/g, "&#39;")
        .replace(/"/g, "&quot;");
    
    return `
        <tr>
            <td>${formatearFecha(dato.timestamp)}</td>
            <td><span class="badge ${getBadgeClass(dato.tipoAccion)}">${dato.tipoAccion}</span></td>
            <td>${dato.usuario || '-'}</td>
            <td>${dato.lob || '-'}</td>
            <td><strong>${dato.bmsId || '-'}</strong></td>
            <td>${dato.nombreAgente || '-'}</td>
            <td style="text-align:right;font-weight:600;color:var(--success);">${monto}</td>
            <td>${dato.mesCDM || '-'}</td>
            <td>
                <button class="btn-primary" 
                        style="padding:6px 12px;font-size:0.85rem;" 
                        onclick='verDetalles(${datosEscapados})'>
                    üëÅÔ∏è Ver
                </button>
            </td>
        </tr>
    `;
}

function formatearFecha(fecha) {
    try {
        return new Date(fecha).toLocaleString('es-CO');
    } catch (e) {
        return 'Fecha inv√°lida';
    }
}

function getBadgeClass(tipoAccion) {
    const classes = {
        'Aprobaci√≥n': 'badge-success',
        'Disputa': 'badge-warning',
        'Traslado': 'badge-info',
        'Prorrateo': 'badge-secondary',
        'Cambio Rol': 'badge-danger'
    };
    return classes[tipoAccion] || 'badge-secondary';
}

function actualizarInfoPaginacion(totalRegistros) {
    const totalPaginas = Math.ceil(totalRegistros / registrosPorPagina);
    
    const elementos = {
        paginaActual: paginaActual,
        totalPaginas: totalPaginas || 1,
        totalRegistros: totalRegistros
    };
    
    Object.entries(elementos).forEach(([id, valor]) => {
        const elemento = document.getElementById(id);
        if (elemento) {
            elemento.textContent = valor;
        }
    });
}

function cambiarPagina(direccion) {
    const totalPaginas = Math.ceil(datosFiltrados.length / registrosPorPagina);
    const nuevaPagina = paginaActual + direccion;
    
    if (nuevaPagina >= 1 && nuevaPagina <= totalPaginas) {
        paginaActual = nuevaPagina;
        renderizarTabla();
        
        const tablaContainer = document.querySelector('.table-container');
        if (tablaContainer) {
            tablaContainer.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }
    }
}

function generarGraficos() {

function destruirGraficos() {
    if (chartTipoAccion) {
        chartTipoAccion.destroy();
        chartTipoAccion = null;
    }
    if (chartActividad) {
        chartActividad.destroy();
        chartActividad = null;
    }
}

function generarGraficoTipoAccion() {
    const ctx = document.getElementById('chartTipoAccion');
    if (!ctx) return;
    
    const contadores = {};
    datosFiltrados.forEach(d => {
        contadores[d.tipoAccion] = (contadores[d.tipoAccion] || 0) + 1;
    });
    
    const labels = Object.keys(contadores);
    const data = Object.values(contadores);
    
    const colores = {
        'Aprobaci√≥n': '#27ae60',
        'Disputa': '#f39c12',
        'Traslado': '#3498db',
        'Prorrateo': '#9b59b6',
        'Cambio Rol': '#e74c3c'
    };
    
    const backgroundColors = labels.map(l => colores[l] || '#95a5a6');
    
    if (chartTipoAccion) chartTipoAccion.destroy();
    
    chartTipoAccion = new Chart(ctx, {
        type: 'doughnut',
        data: {
            labels: labels,
            datasets: [{
                data: data,
                backgroundColor: backgroundColors,
                borderWidth: 2,
                borderColor: '#fff'
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: true,
            plugins: {
                legend: {
                    position: 'bottom'
                },
                tooltip: {
                    callbacks: {
                        label: function(context) {
                            const total = context.dataset.data.reduce((a, b) => a + b, 0);
                            const porcentaje = ((context.parsed / total) * 100).toFixed(1);
                            return `${context.label}: ${context.parsed} (${porcentaje}%)`;
                        }
                    }
                }
            }
        }
    });
}

function generarGraficoActividad() {
    const ctx = document.getElementById('chartActividad');
    if (!ctx) return;
    
    const actividadPorDia = {};
    datosFiltrados.forEach(d => {
        const fecha = new Date(d.timestamp).toLocaleDateString('es-CO');
        actividadPorDia[fecha] = (actividadPorDia[fecha] || 0) + 1;
    });
    
    const fechasOrdenadas = Object.keys(actividadPorDia).sort((a, b) => {
        return new Date(a.split('/').reverse().join('-')) - new Date(b.split('/').reverse().join('-'));
    });
    
    const labels = fechasOrdenadas;
    const data = fechasOrdenadas.map(f => actividadPorDia[f]);
    
    if (chartActividad) chartActividad.destroy();
    
    chartActividad = new Chart(ctx, {
        type: 'line',
        data: {
            labels: labels,
            datasets: [{
                label: 'Operaciones por d√≠a',
                data: data,
                borderColor: '#3498db',
                backgroundColor: 'rgba(52, 152, 219, 0.1)',
                borderWidth: 3,
                fill: true,
                tension: 0.4
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: true,
            plugins: {
                legend: {
                    display: false
                }
            },
            scales: {
                y: {
                    beginAtZero: true,
                    ticks: {
                        stepSize: 1
                    }
                }
            }
        }
    });
}

function generarDescripcion(d) {
    const descripciones = {
        'Aprobaci√≥n': `fue aprobado en <strong>${d.lob}</strong> para el mes de <strong>${d.mesCDM}</strong>`,
        'Disputa': `present√≥ una disputa en <strong>${d.lob}</strong>`,
        'Traslado': `fue trasladado a <strong>${d.lobDestino || 'otra LOB'}</strong>`,
        'Prorrateo': `recibi√≥ prorrateo en <strong>${d.lob}</strong>`,
        'Cambio Rol': `cambi√≥ de rol a <strong>${d.estadoNuevo}</strong>`
    };
    
    return descripciones[d.tipoAccion] || `tuvo una operaci√≥n en <strong>${d.lob}</strong>`;
}

function formatearTiempoRelativo(timestamp) {
    const ahora = new Date();
    const fecha = new Date(timestamp);
    const diferencia = ahora - fecha;
    
    const minutos = Math.floor(diferencia / 60000);
    const horas = Math.floor(diferencia / 3600000);
    const dias = Math.floor(diferencia / 86400000);
    
    if (minutos < 1) return 'Ahora mismo';
    if (minutos < 60) return `Hace ${minutos} min`;
    if (horas < 24) return `Hace ${horas} hora${horas > 1 ? 's' : ''}`;
    if (dias < 30) return `Hace ${dias} d√≠a${dias > 1 ? 's' : ''}`;
    return fecha.toLocaleDateString('es-CO');
}

function obtenerColorEstado(estado) {
    const colores = {
        'Pendiente': '#f39c12',
        '‚è≥ En Proceso': '#3498db',
        '‚úÖ Aprobada': '#27ae60',
        '‚ùå Rechazada': '#e74c3c',
        'üìã M√°s Informaci√≥n': '#9b59b6'
    };
    
    // Buscar coincidencia parcial
    for (const key in colores) {
        if (estado.includes(key) || estado.includes(key.replace(/[‚è≥‚úÖ‚ùåüìã]/g, '').trim())) {
            return colores[key];
        }
    }
    
    return '#95a5a6';
}

function filtrarDisputas() {
    const estadoFiltro = document.getElementById('filtroEstadoDisputa')?.value || '';
    const lobFiltro = document.getElementById('filtroLobDisputa')?.value || '';
    const fechaDesde = document.getElementById('fechaDesdeDisputa')?.value;
    const fechaHasta = document.getElementById('fechaHastaDisputa')?.value;
    const busqueda = document.getElementById('busquedaDisputa')?.value?.toLowerCase() || '';
    
    const fechaDesdeObj = fechaDesde ? new Date(fechaDesde) : null;
    const fechaHastaObj = fechaHasta ? new Date(fechaHasta + 'T23:59:59') : null;
    
    disputasFiltradas = datosDisputas.filter(d => {
        const fechaRegistro = new Date(d.timestamp);
        
        const cumpleEstado = !estadoFiltro || d.estado.includes(estadoFiltro);
        const cumpleLob = !lobFiltro || d.lob === lobFiltro;
        const cumpleFecha = (!fechaDesdeObj || fechaRegistro >= fechaDesdeObj) &&
                           (!fechaHastaObj || fechaRegistro <= fechaHastaObj);
        const cumpleBusqueda = !busqueda ||
                              (d.bmsId && d.bmsId.toString().toLowerCase().includes(busqueda)) ||
                              (d.nombreAgente && d.nombreAgente.toLowerCase().includes(busqueda));
        
        return cumpleEstado && cumpleLob && cumpleFecha && cumpleBusqueda;
    });
    
    paginaActualDisputas = 1;
    renderizarTablaDisputas(disputasFiltradas);
}

function cargarFiltrosDisputas() {
    // ‚≠ê CORREGIDO: Usar LOBs permitidas, NO extraer de datosDisputas
    let lobs;
    
    if (window.lobsPermitidasGlobal && window.lobsPermitidasGlobal.length > 0) {
        lobs = window.lobsPermitidasGlobal.sort();
        console.log(`üìã Cargando ${lobs.length} LOBs permitidas en filtro de disputas:`, lobs);
    } else {
        // Fallback
        lobs = [...new Set(datosDisputas.map(d => d.lob))].sort();
        console.log(`‚ö†Ô∏è Fallback: Usando LOBs de datosDisputas:`, lobs);
    }
    
    const select = document.getElementById('filtroLobDisputa');
    
    if (select) {
        select.innerHTML = '<option value="">Todas las LOBs</option>';
        lobs.forEach(lob => {
            const opt = document.createElement('option');
            opt.value = lob;
            opt.textContent = lob;
            select.appendChild(opt);
        });
    }
}

function limpiarFiltrosDisputas() {
    document.getElementById('filtroEstadoDisputa').value = '';
    document.getElementById('filtroLobDisputa').value = '';
    document.getElementById('fechaDesdeDisputa').value = '';
    document.getElementById('fechaHastaDisputa').value = '';
    document.getElementById('busquedaDisputa').value = '';
    
    filtrarDisputas();
    mostrarToast('üóëÔ∏è Filtros limpiados', 'info');
}

function enviarRespuestaDisputa(event, disputa) {
    event.preventDefault();
    
    const form = event.target;
    const formData = new FormData(form);
    
    const decision = formData.get('decision');
    const respuesta = formData.get('respuesta');
    const montoAjustado = formData.get('montoAjustado');
    const comentarioInterno = formData.get('comentarioInterno');
    
    // Validaciones
    if (!decision || !respuesta) {
        mostrarToast('‚ö†Ô∏è Completa los campos obligatorios', 'warning');
        return;
    }
    
    if (decision === 'APROBADA' && !montoAjustado) {
        mostrarToast('‚ö†Ô∏è Debes ingresar el monto ajustado', 'warning');
        return;
    }
    
    // Confirmar acci√≥n
    const mensajesConfirmacion = {
        'APROBADA': '¬øConfirmas APROBAR esta disputa? Se ajustar√° el bono en Audit.',
        'RECHAZADA': '¬øConfirmas RECHAZAR esta disputa?',
        'MAS_INFO': '¬øConfirmas solicitar m√°s informaci√≥n al agente?',
        'EN_PROCESO': '¬øConfirmas marcar esta disputa como EN PROCESO?'
    };
    
    if (!confirm(mensajesConfirmacion[decision] || '¬øConfirmas esta acci√≥n?')) {
        return;
    }
    
    // Preparar datos
    const datosRespuesta = {
        fila: disputa.fila,
        bmsId: disputa.bmsId,
        nombre: disputa.nombreAgente,
        lob: disputa.lob,
        fechaDisputa: disputa.timestamp,
        estadoOriginal: disputa.estado,
        decision: decision,
        respuesta: respuesta,
        montoAjustado: montoAjustado ? parseFloat(montoAjustado) : null,
        comentarioInterno: comentarioInterno || ''
    };
    
    // Deshabilitar bot√≥n
    const btnSubmit = form.querySelector('button[type="submit"]');
    btnSubmit.disabled = true;
    btnSubmit.innerHTML = '‚è≥ Procesando...';
    
    mostrarToast('‚è≥ Procesando respuesta...', 'info');
    
    // Enviar al backend
    google.script.run
        .withSuccessHandler(function(response) {
            console.log('‚úÖ Respuesta procesada:', response);
            
            if (response.success) {
                mostrarToast(`‚úÖ ${response.mensaje}`, 'success');
                cerrarModalRespuesta();
                
                // Recargar disputas
                cargarDatosIniciales();
            } else {
                mostrarToast(`‚ùå Error: ${response.error}`, 'error');
                btnSubmit.disabled = false;
                btnSubmit.innerHTML = '‚úÖ Enviar Respuesta';
            }
        })
        .withFailureHandler(function(error) {
            console.error('‚ùå Error:', error);
            mostrarToast('‚ùå Error al procesar la respuesta', 'error');
            btnSubmit.disabled = false;
            btnSubmit.innerHTML = '‚úÖ Enviar Respuesta';
        })
        .responderDisputa(datosRespuesta);
}

function exportarDisputas() {
    if (disputasFiltradas.length === 0) {
        mostrarToast('‚ö†Ô∏è No hay disputas para exportar', 'warning');
        return;
    }
    
    try {
        const datosExcel = disputasFiltradas.map(d => ({
            'Timestamp': formatearFecha(d.timestamp),
            'BMS ID': d.bmsId || '',
            'Nombre': d.nombreAgente || '',
            'LOB': d.lob || '',
            'Monto': d.monto || 0,
            'Fecha KPI': d.fechaKPI || '',
            'Comentario': d.detalles || '',
            'Estado': d.estado || '',
            'Usuario': d.usuario || '',
            'Correo': d.correo || ''
        }));
        
        const wb = XLSX.utils.book_new();
        const ws = XLSX.utils.json_to_sheet(datosExcel);
        
        ws['!cols'] = [
            {wch: 20}, {wch: 10}, {wch: 30}, {wch: 15},
            {wch: 15}, {wch: 12}, {wch: 50}, {wch: 20},
            {wch: 25}, {wch: 30}
        ];
        
        XLSX.utils.book_append_sheet(wb, ws, 'Disputas');
        
        const nombreArchivo = `Disputas_${new Date().toISOString().slice(0, 10)}.xlsx`;
        XLSX.writeFile(wb, nombreArchivo);
        
        mostrarToast(`‚úÖ Archivo exportado: ${nombreArchivo}`, 'success');
        
    } catch (error) {
        console.error('Error al exportar:', error);
        mostrarError('Error al exportar disputas: ' + error.message);
    }
}

function descargarJSON(data, nombreArchivo) {

    if (!nuevoTipoAcceso) {

    }
    
    if (!confirm(`¬øCambiar vista a ${nuevoTipoAcceso}?`)) {

        cargarTipoAccesoActual();

    }
    
    mostrarToast('‚è≥ Cambiando vista...', 'info');
    
    const emailUsuario = document.getElementById('correoUsuario').textContent;

        .withSuccessHandler(function(response) {

                setTimeout(() => {
                    window.location.href = `${window.location.origin}${window.location.pathname}?page=${nuevoTipoAcceso}`;

            console.log('Error:', error);
            cargarTipoAccesoActual();
        })
