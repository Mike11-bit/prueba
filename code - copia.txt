    var selectedFile = null;
    var userEmail = '';
    var agentesData = [];

    var selectedAgentData = null;
    window.onload = function () {

    function obtenerEmailUsuario() {
      google.script.run
        .withSuccessHandler(function (email) {
          userEmail = email;
        })
        .obtenerEmailUsuario();
    }
    function cargarAgentes() {
      google.script.run
        .withSuccessHandler(function (agentes) {
          agentesData = agentes;
          var select = document.getElementById('agentSelect');
          select.innerHTML = '<option value="">-- Seleccione un agente --</option>';
          agentes.forEach(function (agente) {
            var option = document.createElement('option');
            option.value = agente.ccm;
            option.text = agente.nombre + ' (CCM: ' + agente.ccm + ')';
            option.setAttribute('data-nombre', agente.nombre);
            option.setAttribute('data-supervisor', agente.supervisor);
            select.appendChild(option);
          });
        })
        .withFailureHandler(function (error) {
          mostrarAlerta('registrar', 'error', 'Error al cargar agentes: ' + error.message);
        })
        .obtenerTodosLosAgentes();
    }
    function mostrarInfoAgente() {
      var select = document.getElementById('agentSelect');
      var selectedOption = select.options[select.selectedIndex];
      if (select.value) {
        var ccm = select.value;
        var nombre = selectedOption.getAttribute('data-nombre');
        var supervisor = selectedOption.getAttribute('data-supervisor');
        selectedAgentData = {
          ccm: ccm,
          nombre: nombre,
          supervisor: supervisor
        };
        document.getElementById('infoAgentCCM').textContent = ccm;
        document.getElementById('infoAgentName').textContent = nombre;
        document.getElementById('infoAgentSupervisor').textContent = supervisor;
        if (selectedFile) {
          var ext = selectedFile.name.split('.').pop();
          var newFileName = nombre + ' - ' + ccm + '.' + ext;
          document.getElementById('infoFileName').textContent = newFileName;
        }
        document.getElementById('selectedAgentInfo').classList.add('show');
      } else {
        selectedAgentData = null;
        document.getElementById('selectedAgentInfo').classList.remove('show');
      }
    }
    function setupFileUpload() {
      var zone = document.getElementById('fileUploadZone');
      var input = document.getElementById('fileInput');
      zone.addEventListener('click', function () {
        input.click();
      });
      input.addEventListener('change', function (e) {
        handleFile(e.target.files[0]);
      });
      zone.addEventListener('dragover', function (e) {
        e.preventDefault();
        zone.classList.add('dragover');
      });
      zone.addEventListener('dragleave', function (e) {
        zone.classList.remove('dragover');
      });
      zone.addEventListener('drop', function (e) {
        e.preventDefault();
        zone.classList.remove('dragover');
        handleFile(e.dataTransfer.files[0]);
      });
    }
    function handleFile(file) {
      if (!file) return;
      var validTypes = ['audio/mpeg', 'audio/mp3', 'video/mp4'];
      var isValidType = validTypes.includes(file.type) || file.name.match(/\.(mp3|mp4)$/i);
      if (!isValidType) {
        mostrarAlerta('registrar', 'error', 'Solo se permiten archivos MP3 o MP4');
        return;
      }
      var maxSize = 50 * 1024 * 1024;
      if (file.size > maxSize) {
        mostrarAlerta('registrar', 'error', 'El archivo no debe superar 50 MB');
        return;
      }
      selectedFile = file;
      document.getElementById('fileName').textContent = file.name;
      document.getElementById('fileSize').textContent = (file.size / (1024 * 1024)).toFixed(2) + ' MB';
      document.getElementById('fileType').textContent = file.type || 'Audio/Video';
      document.getElementById('fileInfo').classList.add('show');
      if (selectedAgentData) {
        var ext = file.name.split('.').pop();
        var newFileName = selectedAgentData.nombre + ' - ' + selectedAgentData.ccm + '.' + ext;
        document.getElementById('infoFileName').textContent = newFileName;
      }
      mostrarAlerta('registrar', 'success', '‚úÖ Archivo cargado correctamente');
    }
    document.getElementById('formRegistro').addEventListener('submit', function (e) {
      e.preventDefault();

    function subirArchivo(agentId) {
      document.getElementById('loadingRegistro').style.display = 'block';
      document.getElementById('btnSubmit').disabled = true;
      document.getElementById('alertRegistro').innerHTML = '';
      var reader = new FileReader();
      reader.onload = function (e) {
        var base64Data = e.target.result.split(',')[1];
        google.script.run
          .withSuccessHandler(function (response) {
            document.getElementById('loadingRegistro').style.display = 'none';
            document.getElementById('btnSubmit').disabled = false;
            if (response && response.success) {
              mostrarAlerta('registrar', 'success', '‚úÖ ' + response.message);
              document.getElementById('formRegistro').reset();
              document.getElementById('fileInfo').classList.remove('show');
              document.getElementById('selectedAgentInfo').classList.remove('show');
              selectedFile = null;
              selectedAgentData = null;
            } else {
              mostrarAlerta('registrar', 'error', '‚ùå ' + (response ? response.message : 'Error desconocido'));
            }
          })
          .withFailureHandler(function (error) {
            document.getElementById('loadingRegistro').style.display = 'none';
            document.getElementById('btnSubmit').disabled = false;
            mostrarAlerta('registrar', 'error', '‚ùå Error: ' + error.message);
          })
          .uploadFile(base64Data, selectedFile.name, selectedFile.type, agentId, userEmail);
      };
      reader.readAsDataURL(selectedFile);
    }

    function mostrarAlerta(tab, tipo, mensaje) {
      var alertDiv = document.getElementById(tab === 'registrar' ? 'alertRegistro' : 'alertConsulta');
      var clase = tipo === 'success' ? 'alert-success' : (tipo === 'error' ? 'alert-error' : 'alert-info');
      alertDiv.innerHTML = '<div class="alert ' + clase + '">' + mensaje + '</div>';
      if (tipo !== 'info') {
        setTimeout(function () {
          alertDiv.innerHTML = '';
        }, 5000);
      }
    }
    function cambiarTab(tabId) {
      var contents = document.getElementsByClassName('tab-content');
      for (var i = 0; i < contents.length; i++) {
        contents[i].classList.remove('active');
      }
      var tabs = document.getElementsByClassName('tab');
      for (var i = 0; i < tabs.length; i++) {
        tabs[i].classList.remove('active');
      }
      document.getElementById(tabId).classList.add('active');
      var allTabs = document.querySelectorAll('.tab');
      allTabs.forEach(function (tab, index) {
        if ((tabId === 'registrar' && index === 0) || (tabId === 'consultar' && index === 1)) {
          tab.classList.add('active');
        }
      });
    }
